solution :1
-----------------------------------------
with max_salary_table as (
	select DEPT_NAME, max(SALARY) AS max_salary 
	from employee
	group by DEPT_NAME 
),
min_salary_table as (
	select DEPT_NAME, min(SALARY) AS min_salary 
	from employee
	group by DEPT_NAME 
)
select e1.* -- ,ms.max_salary, mis.min_salary
from employee e1
left join max_salary_table ms on e1.DEPT_NAME = ms.DEPT_NAME AND e1.SALARY = ms.max_salary
left join min_salary_table mis on e1.DEPT_NAME = mis.DEPT_NAME AND e1.SALARY = mis.min_salary
where ms.max_salary is not null or mis.min_salary is not null
order by e1.DEPT_NAME


solution 2:

select x.*
from employee e
join (select *,
max(salary) over (partition by dept_name) as max_salary,
min(salary) over (partition by dept_name) as min_salary
from employee) x
on e.emp_id = x.emp_id
and (e.salary = x.max_salary or e.salary = x.min_salary)
order by x.dept_name, x.salary;
